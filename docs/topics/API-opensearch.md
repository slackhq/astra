## Preprocessor Node APIs
<api-doc openapi-path="../api/preprocessor_api.yaml">
    <api-endpoint endpoint="/_bulk" method="POST">
        <request>
            <sample lang="JSON" title="JSON example">
                    { "index": {"_index": "testindex", "_id": "1"} }
                    { "field1" : "value1" },
                    { "index": {"_index": "testindex", "_id": "2"} }
                    { "field1" : "value2" }
            </sample>
            <sample lang="JSON" title="JSON expanded">
                    {
                        "index":
                        {
                            "_index": "testindex",
                            "_id": "1"
                        }
                    }
                    {
                        "field1": "value1"
                    }
                     {
                        "index":
                        {
                            "_index": "testindex",
                            "_id": "2"
                        }
                    }
                    {
                        "field1": "value2"
                    }
            </sample>
        </request> 
        <response type="200">
            <sample lang="JSON">
                {
                    "totalDocs": 2,
                    "failedDocs": 0,
                    "errorMsg": ""
                } 
            </sample> 
        </response>
        <response type="400"> 
            <sample lang="JSON">
                {
                    "totalDocs": 0,
                    "failedDocs": 0,
                    "errorMsg": "rate limit exceeded"
                }
            </sample>
        </response>
        <response type="500">
            <sample lang="JSON">
                {
                    "totalDocs": 0,
                    "failedDocs": 0,
                    "errorMsg": "request must contain only 1 unique index"
                }
            </sample>
        </response>
    </api-endpoint>
</api-doc>
